

#include <iostream>

#include <bits/stdc++.h>


using namespace std;
#define ll long long
#define YOUSIF_ALBOUKHARY     ios_base::sync_with_stdio(0),cin.tie(0),cout.tie(0);

int main() {
    YOUSIF_ALBOUKHARY
    int num_of_cards;
    cin>>num_of_cards;
    string s;

    char * sequence = new char [num_of_cards];

    for(int i=0; i<num_of_cards;i++){
        cin>>sequence[i];
        s+=sequence[i];
    }
    stack<char>stk;
    for(int i=0;i<s.length();i++){
        if(stk.empty()){
            stk.push(s[i]);
        }
        else if(stk.top()==s[i]){
            stk.pop();
        } else{
            stk.push(s[i]);
        }
    }
    if(stk.size()<=3)cout<<"Yes"<<'\n';
    else cout<<"No"<<'\n';
    return 0;
}
